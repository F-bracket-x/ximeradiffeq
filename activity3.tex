\item One of the initial value problems below has more than one solution.  Which one is it?

	\begin{enumerate}[label=(\alph*)]
		\item $y^{\,\prime}=\dfrac{x}{y+1}$, $y(0)=0$
		\item $xy^{\,\prime}=\ln{(y)}$, $y(1)=e$
		\item $xy^{\,\prime}=\ln{(y)}$, $y(1)=1$
		\item $y^{\,\prime}=xy^{3/4}$, $y(0)=1$
		\item $y^{\,\prime}=xy^{3/4}$, $y(0)=0$  % ***
        \item $y^{\,\prime}=xy^{4/3}$, $y(0)=1$
		\item $y^{\,\prime}=xy^{4/3}$, $y(0)=0$
	\end{enumerate}

% (e) with options through (g)

\item Consider the IVP $y^{\,\prime}=\dfrac{xy^{1/3}}{y+1}$, $y(0)=a$ where $a$ is a constant.  For how many values of $a$ does the Existence and Uniqueness Theorem \textbf{NOT} guarantee the existence of a unique solution to the IVP (on some open interval)?
    
    % 2 values, -1 and 0

\item Suppose $y(x)$ satisfies $\displaystyle y^{\,\prime}=y-y\ln{\left(y\right)}$, $y(1)=e$.  Find $y(2026)$ to the nearest hundredth.

% e \approx 2.72

\item Let $a$ be a constant.  Suppose $\displaystyle y^{\,\prime}=\frac{\left(y^5-y^3\right)^{1/3}}{1-e^{y^2}}$, $y(0)=a$.  Determine a value of $a$ for which the Existence and Uniquesss Theorem guarantees that a solution exists to this IVP, but where it does not guarantee a unique solution (on some open interval).

\item Suppose $y^{\,\prime}=f(x,y)$ has the slope field below.  There is exactly one value of $0.5<a<5$ for which it is not guaranteed that the IVP $y^{\,\prime}=f(x,y)$, $y(0)=a$ has a unique solution.  Determine $a$ to the nearest tenth.

\begin{figure}[h]
    \centering
    \alttext{A direction field.}
\begin{tikzpicture}
 \begin{axis}[axis lines =middle,
        view={0}{90},
        domain=-2:2,
        y domain=0.6:5.1, % Must be > 0.5 for the Log
        samples=15,
        % Define the slope function and the normalization factor
        declare function={
            f(\x,\y) = ln(\y - 0.5);
            norm(\x,\y) = sqrt(1 + (f(\x,\y))^2);
        }
    ]
        \addplot3 [
            black,thick,
            quiver={
                u={1/norm(x,y)},    % Normalized dx
                v={f(x,y)/norm(x,y)}, % Normalized dy
                scale arrows=0.2,    % Adjust this for arrow size
            },
            -stealth,
        ] {0};
    \end{axis}\end{tikzpicture}
    \caption{A direction field (also known as a slope field)}
\end{figure}
