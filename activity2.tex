\item Which first order differential equation(s) are \textbf{NOT} separable?  Select all that apply.

\begin{enumerate}[label=(\alph*)]
\item $y^{\,\prime}=1-y^2$
\item $y^{\,\prime}=\dfrac{x}{y}+x$
\item $y^{\,\prime}=\dfrac{y}{x}+x$  % ***
\item $y^{\,\prime}=\tan^{-1}(xy)$ % ***
\item $y^{\,\prime}=e^{x^2-y}$
\item $y^{\,\prime}=y^2+xy^2$
\item $y^{\,\prime}=y^3+xy^2$ % ***
\item $y^{\,\prime}=\ln \left(x^y\right)$
\end{enumerate}

% (c), (d), and (g) with options through (h)

\item There is a solution of the form $y=\dfrac{1}{C-x}$ for the first order IVP $y^{\,\prime}=y^2$, $y(1)=0.5$.  Find the constant $C$.

% C=3

\item The differential equation $y^{\,\prime}=2x(y^2-4)\ln (y)$ is separable. How many constant solutions of the form $y=C$ does this differential equation have? 

% 2

\item Find the width of the widest interval on which there exists a solution to the IVP $y^{\,\prime}=-\dfrac{y^2}{x}$, $y(e^{-2})=-1$.  Round to the nearest hundredth.

\item $b$

\item $a+b$

\end{enumerate}
