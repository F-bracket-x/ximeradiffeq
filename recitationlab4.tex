\item  For each of the following second order linear nonhomogeneous differential equations construct your best guess at the form of a particular solution, using the method of Undetermined Coefficients.  You are NOT being asked to solve the equation.
\begin{enumerate}
\item $y^{\,\prime\prime}+y=4t^3+3t^2+2t$.
\vfill

\item $y^{\,\prime\prime}+y=\sin (t)+\cos (t)$.
\vfill


\item $y^{\,\prime\prime}+y=e^{-t}(\sin (t)+\cos (t))$.
\vfill

\item $y^{\,\prime\prime}-y^{\,\prime}-2y=e^{2t}$.
\vfill

\item $y^{\,\prime\prime}-y^{\,\prime}-2y=e^{-2t}$.
\vfill

\item $y^{\,\prime\prime}+6y^{\,\prime}+9y=(t-1)e^{3t}$.
\vfill

\item $y^{\,\prime\prime}+6y^{\,\prime}+9y=(t-1)e^{-3t}$.
\vfill

\item  $y^{\,\prime\prime}+2y^{\,\prime}+10y=-4e^{t}\sin{(3t)}$.\\
\vfill

\item  $y^{\,\prime\prime}+2y^{\,\prime}+10y=-4e^{-t}\sin{(3t)}$.\\
\vfill

\item $y^{\,\prime\prime}+2y^{\,\prime}+10y=\sin{(3t)}$.\\
\vfill

\item $y^{\,\prime\prime\prime}-3y^{\,\prime\prime}+3y^{\,\prime}-y=(1+t^2)e^t$.\\
\vfill

\end{enumerate}
