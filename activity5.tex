\item $\displaystyle y^{\,\prime}=\dfrac{2x}{x+3y}$ 
\item $\displaystyle y^{\,\prime}=\dfrac{2x^2}{x^2+y^2}$
\item $\displaystyle y^{\,\prime}=1+\dfrac{y}{x}$ % ***
\item $\displaystyle y^{\,\prime}=e^{\dfrac{x+y}{x}}$ 
\end{enumerate}

\item Given the nonlinear homogeneous de $\displaystyle y^{\,\prime}=\dfrac{x^2+y^2}{xy}$ determine the transformed separable equation that is the result of applying the substitution $u=\frac{y}{x}$.\\

% u' x + u=1/u+u

 \begin{enumerate}[label=(\alph*)]
\item $\displaystyle u u'=\frac{1}{x}$\\ % ***
\item $\displaystyle \frac{1}{u} u'=\frac{1}{x}$\\
\item $\displaystyle \frac{1}{u^2} u'=-\frac{1}{x}$\\
\item $\displaystyle \frac{u}{1+u^2} u'=\frac{1}{x}$\\
\item $\displaystyle \frac{u^2}{1-u^3} u'=-\frac{1}{x}$\\
